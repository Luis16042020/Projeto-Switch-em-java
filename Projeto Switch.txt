package projetoswich;
import javax.swing.JOptionPane;
/**
 *
 * @author User
 */
public class ProjetoSwich {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
        int mes;
        
        mes = Integer.parseInt(JOptionPane.showInputDialog("Digite um número entre 1 e 12"));
        
        switch (mes){
            case 1:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de janeiro");
                break;
            case 2:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de fevereiro");
                break;
            case 3:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de março");
                break;
            case 4:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de abril");
                break;
            case 5:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de maio");
                break;
            case 6:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de junho");
                break;
            case 7:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de julho");
                break;
            case 8:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de agosto");
                break;
            case 9:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de setembro");
                break;
            case 10:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de outubro");
                break;
            case 11:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de novembro");
                break;
            case 12:
                JOptionPane.showMessageDialog(null,"O número selecionado corresponde ao mês de dezembro");
                break;
            default:
                JOptionPane.showMessageDialog(null,"Número inválido");
                break;
        }
    }
    
}
